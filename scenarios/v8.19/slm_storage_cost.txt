// =====================================================================
// Approach A: Data Stream Test Procedure (For Kibana Dev Tools)
// Target version: Elasticsearch / Elastic Cloud  v8.19
// Variant: SLM Storage Cost Measurement
// =====================================================================
//
// PURPOSE
// -------
// This scenario quantifies the additional repository storage cost of
// retaining SLM snapshots alongside ILM searchable-snapshots.
//
// BACKGROUND:
//   - ILM creates a frozen-tier searchable snapshot of each backing index
//     and keeps it after deletion (delete_searchable_snapshot: false).
//   - SLM can capture the same data independently at controlled moments.
//   - If SLM retention is extended beyond the ILM snapshot lifetime, the
//     SLM snapshot becomes a last-resort recovery source — but at extra cost.
//   - Snapshot repositories use blob-level deduplication: if SLM and ILM
//     snapshots share Lucene segment files, the incremental bytes added
//     by the SLM snapshot can be very small.
//
// THIS SCENARIO ANSWERS:
//   1. How much extra repository storage does an SLM snapshot add on top
//      of the ILM frozen snapshot for the same backing index?
//   2. Does taking SLM AFTER force merge cost less than taking it BEFORE,
//      because the post-merge segments match those in the ILM snapshot?
//
// KEY HYPOTHESIS:
//   SLM snapshot taken AFTER force merge should deduplicate almost
//   completely against the ILM frozen snapshot (both reference the same
//   single-segment Lucene file). SLM taken BEFORE force merge captures a
//   different segment layout → less deduplication → higher extra cost.
//
// TEST FLOW PER BACKING INDEX:
//   Ingest 3 docs (→ multiple segments via separate refresh cycles)
//   ILM hot.rollover fires                     (→ backing index sealed)
//   SLM snapshot Pre-N                         (raw multi-segment state)
//   Manual force merge to 1 segment
//   SLM snapshot Post-N                        (single-segment state)
//   ILM frozen (force_merge_index: false)      (ILM snapshots already-merged data)
//   ILM delete  (backing index removed)
//   Measure with check_snapshot_datasize.sh
//
// Repeated for 5 backing indices (000001–000005).
//
// ⚠️  v8.19 vs v9.2+ DIFFERENCE — NO fm-clone step
// In v9.2+, ILM creates a temporary "fm-clone-<uuid>-" copy of the
// backing index, force-merges THAT clone, and then snapshots the clone.
// In v8.x (including v8.19), ILM snapshots the original backing index
// directly under its original name. Therefore the snapshot contains:
//   .ds-logs-mytest-default-default-2026.05.28-000001
// (no "fm-clone-..." prefix), and restore steps are simpler.
//
// ⚠️  NAMESPACE COLLISION WARNING
// This scenario uses the "default" namespace: logs-mytest-default-default.
// Do NOT run this simultaneously with scenarios/v8.19/base.txt or
// scenarios/v8.19/slm.txt on the same cluster (those use logs-mytest-default,
// and both scenarios register a template matching logs-mytest-*).
// Always confirm no leftover resources before starting.
//
// RESOURCE NAMES
//   Data stream    : logs-mytest-default-default
//   Index template : approach_a_default_template  (pattern: logs-mytest-default-*)
//   ILM policy     : approach_a_default_policy
//   SLM policy     : approach_a_default_slm_policy
//   Restore prefix : restored-default-   (Step 7 optional demo only)

  }

// ---------------------------------------------------------------------
// Step 2: Create ILM policy
// ---------------------------------------------------------------------
// hot:    rollover at 3 docs. With separate doc ingests and a default
//         refresh interval of 1s, each doc lands in its own Lucene segment,
//         giving a meaningful pre-merge vs post-merge storage difference.
//
// frozen: min_age 3m after rollover — provides a ~3-minute window between
//         rollover and ILM's frozen snapshot for manual force merge and
//         SLM snapshots.
//         force_merge_index: false — ILM skips its built-in force merge
//         because we have already merged manually. Both the SLM Post-N
//         snapshot and the ILM frozen snapshot then reference the same
//         single-segment files, maximising blob deduplication.
//
// delete: min_age 8m (3m + 5m buffer) keeps all snapshots available
//         for Step 6 storage measurement after deletion.
PUT _ilm/policy/approach_a_default_policy
{
  "policy": {
    "phases": {
      "hot": {
        "min_age": "0ms",
        "actions": {
          "rollover": { "max_docs": 3 }
        }
      },
      "frozen": {
        "min_age": "3m",
        "actions": {
          "searchable_snapshot": {
            "snapshot_repository": "found-snapshots",
            "force_merge_index": false
          }
        }
      },
      "delete": {
        "min_age": "8m",
        "actions": {
          "delete": {
            "delete_searchable_snapshot": false
          }
        }
      }
    }
  }
}

// ---------------------------------------------------------------------
// Step 2.5: Create SLM policy
// ---------------------------------------------------------------------
// Far-future schedule (year 2099): never fires automatically.
// All snapshots are executed manually at precise measurement checkpoints.
//
// max_count: 20 — retains all SLM snapshots for this test session.
// Pre + Post for 5 backing indices = 10 SLM snapshots; 20 leaves headroom.
//
// partial: true — allows the snapshot to complete even if a partial mount
// (frozen backing index) is present in the data stream at snapshot time.
PUT _slm/policy/approach_a_default_slm_policy
{
  "name": "<approach-a-storage-cost-{now/d}>",
  "schedule": "0 0 0 1 1 ? 2099",
  "repository": "found-snapshots",
  "config": {
    "indices": ["logs-mytest-default-default"],
    "include_global_state": false,
    "partial": true
  },
  "retention": {
    "expire_after": "30d",
    "min_count": 1,
    "max_count": 20
  }
}

// Verify the policy was created:
GET _slm/policy/approach_a_default_slm_policy?human
