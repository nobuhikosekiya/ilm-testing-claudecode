// =====================================================================
// Approach A: Data Stream Test Procedure (For Kibana Dev Tools)
// Target version: Elasticsearch / Elastic Cloud  v8.19
// Variant: WITH SLM (Snapshot Lifecycle Management) Demonstration
// =====================================================================
//
// PURPOSE OF THIS VARIANT
// ========================
// This scenario extends the base ILM-restore test to show that SLM
// is NOT a reliable recovery mechanism for Frozen-tier backing indices.
// The key insight:
//   - SLM snapshots capture indices at the time of execution.
//   - Once ILM deletes a backing index AND SLM retention rotates out
//     the snapshots that captured it, that data is unrecoverable via SLM.
//   - ILM-managed searchable-snapshot retention
//     (delete_searchable_snapshot: false) is the correct path.
//
// ⚠️  v8.19 vs v9.2+ DIFFERENCE — NO fm-clone step
// In v9.2+, ILM creates a temporary "fm-clone-<uuid>-" copy of the
// backing index, force-merges THAT clone, and then snapshots the clone.
// In v8.x (including v8.19), ILM performs the force merge directly on
// the original backing index and snapshots it under its original name.
// Therefore the snapshot contains:
//   .ds-logs-mytest-default-2026.05.27-000001
// (no "fm-clone-..." prefix), and the restore steps below are simpler.

// ---------------------------------------------------------------------
// Step 1: Preparation (Shorten ILM poll interval)
// ---------------------------------------------------------------------
PUT _cluster/settings
{
  "transient": {
    "indices.lifecycle.poll_interval": "10s"
  }
}

// ---------------------------------------------------------------------
// Step 2: Create ILM policy
// ---------------------------------------------------------------------
// Policy triggers rollover at 1 document, moves to Frozen,
// and deletes the backing index after 3 minutes while keeping the snapshot.
PUT _ilm/policy/approach_a_ds_policy
{
  "policy": {
    "phases": {
      "hot": {
        "actions": {
          "rollover": { "max_docs": 1 }
        }
      },
      "frozen": {
        "min_age": "0m",
        "actions": {
          "searchable_snapshot": {
            "snapshot_repository": "found-snapshots"
          }
        }
      },
      "delete": {
        "min_age": "3m",
        "actions": {
          "delete": {
            "delete_searchable_snapshot": false
          }
        }
      }
    }
  }
}

// ---------------------------------------------------------------------
// Step 2.5: Create SLM policy
// ---------------------------------------------------------------------
// We schedule the SLM policy with a far-future cron so it never fires
// automatically — we will execute it manually at precise test moments.
//
// KEY SETTING: max_count: 3
//   SLM will retain only the 3 newest snapshots. Once we run more than
//   3 executions, the oldest will be deleted by retention. This is the
//   core mechanism that makes SLM unreliable for long-term frozen recovery.
//
// partial: true — allows the snapshot to complete even if a backing index
// is momentarily unavailable during a rollover transition.
PUT _slm/policy/approach_a_slm_policy
{
  "name": "<approach-a-slm-{now/d}>",
  "schedule": "0 0 0 1 1 ? 2099",
  "repository": "found-snapshots",
  "config": {
    "indices": "*",
    "include_global_state": false,
    "partial": true
  },
  "retention": {
    "expire_after": "30d",
    "min_count": 1,
    "max_count": 3
  }
}

// Verify the policy was created:
GET _slm/policy/approach_a_slm_policy?human

// ---------------------------------------------------------------------
// Step 3: Create Index Template for Data Streams
// ---------------------------------------------------------------------
// Notice the "data_stream": { } block. This tells Elasticsearch to
// treat any index matching "logs-mytest-*" as a Data Stream.
// IMPORTANT: "priority": 200 is required to outrank the built-in "logs"
// template (priority: 100, pattern: logs-*-*) which would otherwise
// capture this data stream and apply the wrong ILM policy.
PUT _index_template/approach_a_ds_template
{
  "index_patterns": ["logs-mytest-*"],
  "priority": 200,
  "data_stream": { },
  "template": {
    "settings": {
      "index.lifecycle.name": "approach_a_ds_policy"
    }
  }
}

// ---------------------------------------------------------------------
// Step 4: Ingestion, SLM Snapshots, & ILM Rollovers
// ---------------------------------------------------------------------
// IMPORTANT SEQUENCE for each document:
//   1. Ingest the document (backing index is created / written to).
//   2. Execute SLM immediately — BEFORE the ILM rollover fires.
//      This ensures the snapshot captures the backing index while it
//      is still in the hot tier and fully present on the cluster.
//   3. Wait for ILM to roll over (creates the next backing index).

// --- 4.1 Ingest Doc 1 (Goes into backing index 000001) ---
POST logs-mytest-default/_doc
{
  "@timestamp": "2026-05-27T10:00:00Z",
  "message": "Log event 1 in backing index 1"
}

// Execute SLM snapshot #1 BEFORE ILM rolls over.
// This snapshot will contain .ds-logs-mytest-default-2026.05.27-000001.
POST _slm/policy/approach_a_slm_policy/_execute

// Confirm SLM snapshot 1 completed (look for last_success.snapshot_name):
GET _slm/policy/approach_a_slm_policy?human

// WAIT ~15 SECONDS for ILM to evaluate the max_docs: 1 condition
// and automatically roll over to a new backing index.

// --- 4.2 Ingest Doc 2 (Goes into backing index 000002) ---
POST logs-mytest-default/_doc
{
  "@timestamp": "2026-05-27T10:05:00Z",
  "message": "Log event 2 in backing index 2"
}

// Execute SLM snapshot #2 BEFORE ILM rolls over.
// This snapshot will contain 000001 (now frozen) and 000002 (still hot).
POST _slm/policy/approach_a_slm_policy/_execute

// Confirm SLM snapshot 2 completed:
GET _slm/policy/approach_a_slm_policy?human

// WAIT ~15 SECONDS for the next ILM rollover.

// --- 4.3 Ingest Doc 3 (Goes into backing index 000003) ---
POST logs-mytest-default/_doc
{
  "@timestamp": "2026-05-27T10:10:00Z",
  "message": "Log event 3 in backing index 3"
}

// Execute SLM snapshot #3.
// This snapshot will contain 000001, 000002 (frozen or in transition), and 000003 (hot).
POST _slm/policy/approach_a_slm_policy/_execute

// Confirm SLM snapshot 3 completed:
GET _slm/policy/approach_a_slm_policy?human

// All snapshots to date — you should see exactly 3 SLM snapshots:
GET _snapshot/found-snapshots/_all?sort=start_time&order=desc

// ---------------------------------------------------------------------
// Step 5: Identify Backing Indices & Wait for Deletion
// ---------------------------------------------------------------------
// Check the actual names of the hidden backing indices.
// They will look like: .ds-logs-mytest-default-2026.05.27-000001
GET _cat/indices/.ds-logs-mytest-*?v

// Wait a few minutes. Check the data stream status until you see the
// older backing indices drop off (deleted by ILM) and return 404.
GET .ds-logs-mytest-default-2026.05.27-000001

// ---------------------------------------------------------------------
// Step 6: 👉 BASH SCRIPT INTEGRATION (Restore Phase — ILM snapshots)
// ---------------------------------------------------------------------
// AT THIS POINT IN REAL LIFE: You know you need data from backing index 1
// and backing index 2. Since they are deleted, you must restore them one by one.

// [BASH SCRIPT USAGE]: Run `./scripts/v8.19/restore_index.sh` targeting the exact
// backing index name (e.g., TARGET_INDEX=".ds-logs-mytest-default-2026.05.27-000001").

// ℹ️  v8.19 Snapshot Index Naming:
// In v8.x, ILM force-merges the backing index in place before snapshotting.
// The snapshot therefore contains the original backing index name:
//   .ds-logs-mytest-default-2026.05.27-000001
// (no "fm-clone-..." prefix).
//
// To find the snapshot name and confirm the index name inside it, run:
//   GET _snapshot/found-snapshots/<ilm_snapshot_name>
// and look at the "indices" array — the entry will be the raw ".ds-..." name.
//
// The rename_pattern below uses "\.ds-(.+)" to strip the ".ds-" prefix
// and restore as a clean "restored-..." index.

// [MANUAL DEV TOOLS EQUIVALENT - RESTORING INDEX 1 from ILM snapshot]:
// 1. Find ILM snapshots (named: <date>-<backing-index-name>-<policy>-<uuid>):
GET _snapshot/found-snapshots/_all?sort=start_time&order=desc

// 2. Inspect the snapshot to confirm the index name:
// GET _snapshot/found-snapshots/<ilm_snapshot_name_for_000001>
// Look for "indices": [".ds-logs-mytest-default-2026.05.27-000001"]
// (no fm-clone prefix in v8.19)

// 3. Execute the restore for Backing Index 1.
POST _snapshot/found-snapshots/<ilm_snapshot_name_for_000001>/_restore
{
  "indices": ".ds-logs-mytest-default-2026.05.27-000001",
  "rename_pattern": "\\.ds-(.+)",
  "rename_replacement": "restored-$1",
  "index_settings": {
    "index.lifecycle.name": null,
    "index.routing.allocation.include._tier_preference": "data_hot"
  }
}

// [MANUAL DEV TOOLS EQUIVALENT - RESTORING INDEX 2 from ILM snapshot]:
POST _snapshot/found-snapshots/<ilm_snapshot_name_for_000002>/_restore
{
  "indices": ".ds-logs-mytest-default-2026.05.27-000002",
  "rename_pattern": "\\.ds-(.+)",
  "rename_replacement": "restored-$1",
  "index_settings": {
    "index.lifecycle.name": null,
    "index.routing.allocation.include._tier_preference": "data_hot"
  }
}

// ---------------------------------------------------------------------
// Step 7: Querying the ILM-Restored Indices
// ---------------------------------------------------------------------
// ⚠️  IMPORTANT — Restored indices inherit index.hidden = true from the
// backing index source. You must set it to false before wildcard searches work.
PUT restored-logs-mytest-default-2026.05.27-000001/_settings
{ "index": { "hidden": false } }

PUT restored-logs-mytest-default-2026.05.27-000002/_settings
{ "index": { "hidden": false } }

// Query the ILM-restored indices — both docs should be found.
GET restored-logs-mytest-*/_search

// ---------------------------------------------------------------------
// Step 7.5: Restore from SLM Snapshot (should SUCCEED at this point)
// ---------------------------------------------------------------------
// At this moment the 3 SLM snapshots still exist (we have only run 3
// executions, and max_count=3, so none have been deleted yet).
// The oldest SLM snapshot captured .ds-logs-mytest-default-2026.05.27-000001
// when it was still in the hot tier — we can restore from it.

// 1. List all snapshots and identify the oldest SLM snapshot
//    (named approach-a-slm-<date>-<uuid>):
GET _snapshot/found-snapshots/_all?sort=start_time&order=asc

// 2. Inspect the oldest SLM snapshot to confirm it contains 000001:
// GET _snapshot/found-snapshots/<oldest_slm_snapshot>
// You should see ".ds-logs-mytest-default-2026.05.27-000001" in "indices".

// 3. Restore 000001 from SLM snapshot, using a different prefix ("restored-slm-")
//    so it does not collide with the ILM-restored index from Step 6.
POST _snapshot/found-snapshots/<oldest_slm_snapshot>/_restore
{
  "indices": ".ds-logs-mytest-default-2026.05.27-000001",
  "rename_pattern": "\\.ds-(.+)",
  "rename_replacement": "restored-slm-$1",
  "index_settings": {
    "index.lifecycle.name": null,
    "index.routing.allocation.include._tier_preference": "data_hot"
  }
}

// Unhide the SLM-restored index:
PUT restored-slm-logs-mytest-default-2026.05.27-000001/_settings
{ "index": { "hidden": false } }

// Query — should return "Log event 1 in backing index 1":
GET restored-slm-logs-mytest-*/_search

// ✅ SUCCESS: At this point SLM restore works because we haven't yet
//    rotated out the snapshot that captured 000001.

// ---------------------------------------------------------------------
// Step 7.6: Demonstrate SLM Retention Breaking Recovery
// ---------------------------------------------------------------------
// Now run SLM 3 more times. Total executions will be 6.
// With max_count=3, SLM retention MUST delete the 3 oldest snapshots
// (the ones that contained .ds-logs-mytest-default-2026.05.27-000001).

// Run 3 additional SLM executions:
POST _slm/policy/approach_a_slm_policy/_execute
// (wait a few seconds between each to get distinct snapshot names)
POST _slm/policy/approach_a_slm_policy/_execute
POST _slm/policy/approach_a_slm_policy/_execute

// Trigger SLM retention immediately (it does not run synchronously
// after _execute — this forces it to run now):
POST _slm/_execute_retention

// Wait ~10 seconds, then verify only 3 snapshots remain.
// The 3 oldest (approach-a-slm-*) should be GONE:
GET _snapshot/found-snapshots/_all?sort=start_time&order=desc

// ⚠️  Now attempt to restore 000001 from the NEWEST SLM snapshot.
// The newest snapshot was taken after ILM deleted 000001 from the cluster,
// so it cannot contain that index. The snapshots that DID contain it
// have already been deleted by retention.
POST _snapshot/found-snapshots/<newest_slm_snapshot>/_restore
{
  "indices": ".ds-logs-mytest-default-2026.05.27-000001",
  "rename_pattern": "\\.ds-(.+)",
  "rename_replacement": "restored-slm-late-$1",
  "index_settings": {
    "index.lifecycle.name": null,
    "index.routing.allocation.include._tier_preference": "data_hot"
  }
}

// ❌ EXPECTED FAILURE:
// {
//   "error": {
//     "type": "index_not_found_exception",
//     "reason": "no such index [.ds-logs-mytest-default-2026.05.27-000001]",
//     "index": ".ds-logs-mytest-default-2026.05.27-000001"
//   },
//   "status": 404
// }
//
// CONCLUSION:
// In a Frozen-tier ILM workflow, SLM snapshots are NOT a reliable recovery
// mechanism for deleted backing indices. Once ILM deletes a backing index
// AND SLM retention rotates past the snapshots that captured it, that
// data is unrecoverable via SLM.
//
// ILM-managed searchable-snapshot retention with
//   "delete_searchable_snapshot": false
// is the correct and reliable long-term recovery path for frozen indices.

// ---------------------------------------------------------------------
// Step 8: 👉 BASH SCRIPT INTEGRATION (Cleanup Phase)
// ---------------------------------------------------------------------
// [BASH SCRIPT USAGE]: Same as before, schedule `./scripts/v8.19/cleanup_old_snapshots.sh`
// to delete ILM snapshots older than 2 years.

// [MANUAL DEV TOOLS EQUIVALENT]:
DELETE _snapshot/found-snapshots/<ilm_snapshot_name_for_000001>
DELETE _snapshot/found-snapshots/<ilm_snapshot_name_for_000002>

// ---------------------------------------------------------------------
// Step 9: Environment Cleanup
// ---------------------------------------------------------------------
PUT _cluster/settings
{
  "transient": {
    "indices.lifecycle.poll_interval": null
  }
}

// Delete the SLM policy:
DELETE _slm/policy/approach_a_slm_policy

// Delete remaining SLM snapshots from this test (use actual names from Step 7.6):
// DELETE _snapshot/found-snapshots/<slm_snapshot_4>
// DELETE _snapshot/found-snapshots/<slm_snapshot_5>
// DELETE _snapshot/found-snapshots/<slm_snapshot_6>

// Delete the data stream, its remaining active backing indices, and all restored data:
DELETE _data_stream/logs-mytest-default
DELETE restored-logs-mytest-*
DELETE restored-slm-logs-mytest-*
